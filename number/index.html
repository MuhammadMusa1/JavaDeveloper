5. Коллекции

ArrayList

import java.util.ArrayList;

ArrayList<String> cars = new ArrayList<>();
cars.add("Volvo");
cars.add("BMW");
System.out.println(cars.get(0)); // "Volvo"

HashMap

import java.util.HashMap;

HashMap<String, String> capitalCities = new HashMap<>();
capitalCities.put("England", "London");
System.out.println(capitalCities.get("England")); // "London"

6. Исключения (try-catch)

try {
    int[] myNumbers = {1, 2, 3};
    System.out.println(myNumbers[10]); // Вызовет исключение
} catch (Exception e) {
    System.out.println("Что-то пошло не так: " + e.getMessage());
} finally {
    System.out.println("Блок finally всегда выполняется.");
}
